\begin{frame}
  \frametitle{Bitcoin Scripting Language (``Script'')}
  
  \begin{itemize}
  	\item Built for Bitcoin (inspired by Forth). 
	\item Simple, compact.
	\item Support for cryptography.
	\item Stack-based.
	\item Limits on time/memory.
	\item No looping.

  \end{itemize}
  
\end{frame}
